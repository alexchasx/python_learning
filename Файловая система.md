
# Примеры работ с файловой системой

## Применяемые модули
- os — базовые операции с файловой системой, работа с путями, управление директориями.
- shutil — высокоуровневые операции: копирование, перемещение, архивирование файлов и директорий.
- pathlib — современный объектно‑ориентированный подход (рекомендуется для новых проектов).
- open() — чтение и запись файлов.

## 1. Базовые операции с файлами (модуль os)

```py
import os

# Создание директории
os.mkdir('new_folder')  # Создаёт одну директорию

# Создание вложенной структуры директорий
os.makedirs('parent/child/grandchild', exist_ok=True)

# Проверка существования пути
if os.path.exists('new_folder'):
    print("Директория существует")

# Получение содержимого директории
files = os.listdir('.')
print("Файлы и папки в текущей директории:", files)

# Удаление файла
os.remove('temp.txt')

# Переименование файла или директории
os.rename('old_name.txt', 'new_name.txt')

# Смена текущей рабочей директории
os.chdir('/path/to/directory')

# Получение текущей рабочей директории
current_dir = os.getcwd()
print("Текущая директория:", current_dir)

```

## 2. Рекурсивный обход директорий (os.walk)

```py
import os

for root, dirs, files in os.walk('.'):
    for file in files:
        full_path = os.path.join(root, file)
        print(full_path)

```

## 3. Копирование и перемещение файлов (shutil)

```py
import shutil
import os

# Копирование файла (без метаданных)
shutil.copy('file.txt', 'backup/file.txt')

# Полное копирование с сохранением атрибутов
shutil.copy2('file.txt', 'backup/file.txt')

# Копирование директории со всем содержимым
shutil.copytree('project', 'project_backup')

# Перемещение файла или директории
shutil.move('source/file.txt', 'destination/file.txt')

# Создание новой директории для перемещения (если её нет)
src = 'data/report.txt'
dst_dir = 'archive'
os.makedirs(dst_dir, exist_ok=True)
shutil.move(src, os.path.join(dst_dir, 'report.txt'))

```

## 4. Современный подход с pathlib (Python 3.4+)

```py
from pathlib import Path

# Создание директории
path = Path('data')
path.mkdir(parents=True, exist_ok=True)

# Перемещение файла
src = Path('report.txt')
dst = path / 'report.txt'
src.rename(dst)

# Поиск файлов по шаблону
for file in Path("texts").glob("*.txt"):
    shutil.copy(file, Path("backup") / file.name)

# Проверка существования и типа пути
if path.exists():
    print("Путь существует")
if path.is_dir():
    print("Это директория")
if path.is_file():
    print("Это файл")

```

## 5. Чтение и запись файлов

```py
# Запись в файл
with open('log.txt', 'w') as f:
    f.write("Первая строка\n")
    f.write("Вторая строка\n")

# Чтение файла целиком
with open('log.txt', 'r') as f:
    content = f.read()
    print(content)

# Построчное чтение файла
with open('log.txt', 'r') as f:
    for line in f:
        print(line.strip())

# Запись списка строк
lines = ["Строка 1\n", "Строка 2\n"]
with open('data.txt', 'w') as f:
    f.writelines(lines)

```

## 6. Обработка ошибок

```py
import shutil
import os

try:
    shutil.move('source_file.txt', 'destination/file.txt')
except FileNotFoundError:
    print("Исходный файл не найден")
except PermissionError:
    print("Недостаточно прав для выполнения операции")
except OSError as e:
    print(f"Ошибка операционной системы: {e}")

```
